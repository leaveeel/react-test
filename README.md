> // 不使用exact属性，在/home页面同时展示/,/home内容
> <Route path='/' component={} />
> <Route path='/home' component={home} />
>
> // 使用exact属性，在/home页面只展示/home内容
> <Route exact path='/' component={} />
> <Route path='/home' component={home} />
